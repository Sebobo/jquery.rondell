Preset - Gallery


This demo fetches links &amp; images from my brothers great webservice wookmark.com and creates a rondell from the data.

The image titles are used as caption and clicking the focused element will open the wookmark in a new tab or window.

Here's how it's done

var galleryContainer = $("#rondellGallery");

$.ajax({
  url: 'http://www.wookmark.com/api/json',
  dataType: 'jsonp',
  success: function(json, text, xhqr) {
    if (text == 'success') {
      var items = json, newItem, newImg;

      // Parsing returned json data
      $.each(items, function(idx, item) {
        if (idx >= 20) return false;

        // Create link container
        newItem = $('<a/>')
        .css('opacity', 0)
        .attr({
          href: item.url,
          title: item.title,
          target: '_blank'
        });

        // Create image
        newImg = $('<img/>')
        .addClass('resizeable')
        .attr({
          src: item.preview,
          alt: item.title,
          width: item.width,
          height: item.height
        });

        // Append image to link and link to container for the rondell
        newItem.append(newImg);
        galleryContainer.append(newItem);
      });

      // Create gallery 
      $('a', galleryContainer).rondell({
        preset: 'gallery'                             
      });
    }
  }
});
